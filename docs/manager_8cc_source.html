218  void Manager::update_elapsed_time() {
219 
220  // Only run in simulated time if we have the flag set and have processes.
221  if(_simulated_time && !_processes.empty()) {
222 
223  // Find the process that has the smallest time till next update.
224  auto min_iter = std::min_element(_processes.begin(), _processes.end() ,[](Process * lhs, Process * rhs) {
225  auto lhsTime = lhs->last_update() + lhs->period();
226  auto rhsTime = rhs->last_update() + rhs->period();
227  return lhsTime < rhsTime;
228  });
229 
230  Process* nextup = *min_iter;

233
