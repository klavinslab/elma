213  void Manager::update_elapsed_time() {
214 
215  // Only run in simulated time if we have the flag set and have processes.
216  if(_simulated_time && !_processes.empty()) {
217 
218  // Find the process that has the smallest time till next update.
219  auto min_iter = std::min_element(_processes.begin(), _processes.end() ,[](Process * lhs, Process * rhs) {
220  auto lhsTime = lhs->last_update() + lhs->period();
221  auto rhsTime = rhs->last_update() + rhs->period();
222  return lhsTime < rhsTime;
223  });
224 
225  Process* nextup = *min_iter;
226 
227  auto newTime = nextup->last_update() + nextup->period();
228 
229  _elapsed = newTime;
230 
231  } else {
